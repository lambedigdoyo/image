Rotate Images Based on Changes in Input Values

Rotate an image around its center in canvas

Change the input values ​​below to play rotate:

<input type=number id=degree value=175 step=5 style='width: 50px; display: block; margin: 10px auto 0'/>
<div class=cvsBox>  
  <canvas id=cvs width='416' height='416' style='background-color: lightblue'></canvas>
</div>

<script>
var cvs = document.getElementById('cvs');
var ctx = cvs.getContext('2d');
var degree = document.getElementById('degree');

var image = new Image();
    image.src = '../../../lib/main/images/GR-cah-ndase.png';
function putar(){
    var degrees = degree.value;
    ctx.clearRect(0, 0, cvs.width, cvs.height);
    ctx.save();
    ctx.translate(cvs.width/2, cvs.height/2);
    ctx.rotate(degrees*Math.PI/180);
    ctx.drawImage(image,-image.width/2,-image.height/2);    
    ctx.restore();
}
image.onload = putar;
degree.oninput = putar;
</script>
