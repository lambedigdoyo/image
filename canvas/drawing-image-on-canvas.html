HTML Canvas Images

Drawing images on canvas

<canvas id='canvasImage' width='300' height='200'></canvas>

<script>
var canvas = document.getElementById("canvasImage");
var ci = canvas.getContext("2d");
var image = new Image();
image.src = "https://raw.githubusercontent.com/lambedigdoyo/lib/main/images/bikini.png";
image.onload = function() {
  ci.drawImage(image, 85, 25); // (image, x, y)
}
</script>
