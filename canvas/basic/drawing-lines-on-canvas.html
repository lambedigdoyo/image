Drawing a Line

Drawing Lines on HTML5 Canvas

context.moveTo(35, 125); // context.moveTo(x, y) : start point
context.lineTo(375, 125); // context.lineTo(x, y) : end point
<style>
canvas {
  background: lightyellow;
  border-radius: 6px;
  display: block;
  margin: 10px auto;
  border: 6px solid #aaa;
}
</style>

<div align=center>
  <canvas id=cvs width='400' height='250'></canvas>
</div>

<script>
var cvs = document.getElementById('cvs');

// line-A
var ctx = cvs.getContext("2d");
ctx.moveTo(35, 125); // ctx.moveTo(x, y) : start point
ctx.lineTo(375, 125); // ctx.lineTo(x, y) : end point
ctx.stroke();

// line-B
ctx.beginPath();
ctx.moveTo(100, 200);
ctx.lineTo(300, 200);
ctx.strokeStyle='blue';
ctx.stroke();

// line-C
ctx.beginPath();
ctx.moveTo(200, 75);
ctx.lineTo(200, 170);
ctx.strokeStyle='#0ff';
ctx.stroke();

// line-D
ctx.beginPath();
ctx.moveTo(50, 215);
ctx.lineTo(350, 35);
ctx.lineWidth ='10';
ctx.strokeStyle='#f0f';
ctx.stroke();
</script>
