Rotate an Image

Rotate an image in canvas.

<canvas id=canvas width='400' height='220' style='display: block; background-color: lightblue; margin: 10px auto'></canvas>

<script>
var canvas = document.getElementById('canvas');

var ctx=canvas.getContext("2d");
ctx.fillStyle='black';
ctx.globalAlpha=1.0;        
var rect={ x:100, y:100, width:200, height:20 };
// black : first image
ctx.fillRect( rect.x, rect.y, rect.width, rect.height );

// red: image rotate 90deg (Math.PI/2) from black image
ctx.translate( rect.x+rect.width/2, rect.y+rect.height/2 );
ctx.rotate(Math.PI/2);
ctx.fillStyle='red';
ctx.translate( -rect.x-rect.width/2, -rect.y-rect.height/2 );
ctx.fillRect( rect.x, rect.y, rect.width, rect.height );

// green: rotate 45deg (Math.PI/4) from red image
ctx.translate( rect.x+rect.width/2, rect.y+rect.height/2 );
ctx.rotate(Math.PI/4);
ctx.fillStyle='green';
ctx.translate( -rect.x-rect.width/2, -rect.y-rect.height/2 );
ctx.fillRect( rect.x, rect.y, rect.width, rect.height );

// blue: rotate 60deg (Math.PI/3) from green
// ctx.clearRect(0, 0,canvas.width, canvas.height);
// ctx.save();
ctx.translate( rect.x+rect.width/2, rect.y+rect.height/2 );
ctx.rotate(Math.PI/3);
ctx.fillStyle='blue';
ctx.translate( -rect.x-rect.width/2, -rect.y-rect.height/2 );
ctx.fillRect( rect.x, rect.y, rect.width, rect.height );
//ctx.restore();
</script>
