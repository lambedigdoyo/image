<script>
var canvas = document.getElementById('cvs');
var ctx = canvas.getContext('2d');

// A
ctx.arc(80, 80, 70, 0, 2*Math.PI);
ctx.stroke();

// B: center a drawing
var centerX = canvas.width / 2;
var centerY = canvas.height / 2;
r  = 100;

ctx.beginPath();
ctx.arc(centerX, centerY, r, 0, 2 * Math.PI, false);
ctx.fillStyle='#4fe5fe';
ctx.lineWidth = 4;
ctx.strokeStyle="#00f";
ctx.fill();
ctx.stroke();

// C
ctx.beginPath();
ctx.fillStyle = 'magenta';
ctx.arc(340, 200, 40, 0, 2*Math.PI);
ctx.fill();
</script>
