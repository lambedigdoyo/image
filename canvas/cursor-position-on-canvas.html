Cursor Positions on Canvas

Get the mouse coordinates onmousemove

<style>
canvas {
  background: lightyellow;
  border-radius: 6px;
  display: block;
  margin: 10px auto;
  border: 6px solid #aaa;
}
</style>

<div align=center>
  <canvas id=cvid width='400' height='250'></canvas>
</div>

<script>
var cv = document.getElementById('cvid');
var ctx = cv.getContext("2d");
var ctx2 = cv.getContext("2d");
ctx.font = "16px Arial";
var centerX = cv.width / 2;
var centerY = cv.height / 2;
var gBCR = cv.getBoundingClientRect();

cv.addEventListener("mousemove", function(e) {
  var cvX = Math.round(e.clientX - gBCR.left);
  var cvY = Math.round(e.clientY - gBCR.top);
  ctx2.clearRect(0, 0, cv.width, cv.height);
  drawEmoji(); 
  ctx.fillStyle = 'black';
  ctx.fillText("X: "+cvX+", Y: "+cvY, centerX-50, 20);
});

function drawEmoji() {
  ctx2.beginPath();
  ctx2.fillStyle = 'red';
  ctx2.arc(centerX, centerY, 5, 0, 2*Math.PI);
  ctx2.fill();
  ctx2.stroke(); 
  ctx.closePath();

  ctx.beginPath();
  ctx.lineWidth = '3';   
  ctx.arc(200, 125, 65, 0, Math.PI * 2, true); // Outer circle
  ctx.moveTo(110, 95);
  ctx.stroke();
  ctx.beginPath();
  ctx.lineWidth = '8';
  ctx.strokeStyle ='red';
  ctx.arc(200, 120, 57, 0.35, 0.9*Math.PI, false); // Mouth  
  ctx.stroke();
  ctx.closePath();
    
  ctx.beginPath();
  ctx.lineWidth = '3';    
  ctx.strokeStyle ='black';
  ctx.fillStyle = 'orange';   
  ctx.moveTo(155, 115);
  ctx.arc(175, 115, 18, 0, Math.PI * 2, true); // Left eye
  ctx.moveTo(205, 115);
  ctx.arc(225, 115, 18, 0, Math.PI * 2, true); // Right eye
  ctx.fill();    
  ctx.stroke();
  ctx.closePath();
}
drawEmoji()
</script>
