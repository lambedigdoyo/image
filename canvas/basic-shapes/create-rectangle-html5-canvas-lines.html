Canvas Shapes- Canvas Line Drawing

<canvas id='cvs1' width='200' height='150'></canvas>

<script>
var canvas = document.getElementById("cvs2");
var ct2 = canvas.getContext("2d");
ct2.strokeStyle = "red"; //color of the line
ct2.moveTo(20, 20); //starting point of line1 (x1,y1)
ct2.lineTo(180, 20); // end point of line1 (x2,y2)
ct2.lineTo(180, 130); // end point of line2 (x3,y3)
ct2.lineTo(20, 130); // end point of line3 (x4,y4)
ct2.lineTo(20, 20); // end point of line4 (x1,y1) -  back to start
ct2.lineWidth = 3;
ct2.stroke();
</script>
