<script>
var canvas = document.getElementById("cvs8");
var ct8 = canvas.getContext("2d");
ct8.strokeStyle = "green"; //color of the line
ct8.moveTo(70, 10); //starting point of line1 (x1,y1)
ct8.lineTo(130, 10); // end point of line1 (x2,y2)
ct8.lineTo(165, 50); // end point of line2 (x3,y3)
ct8.lineTo(165, 105); // end point of line3 (x4,y4)
ct8.lineTo(130, 140); // end point of line4 (x5,y5)
ct8.lineTo(70, 140); // end point of line5 (x6,y6) 
ct8.lineTo(35, 105); // end point of line6 (x7,y7)
ct8.lineTo(35, 50); // end point of line7 (x8,y8)
ct8.lineTo(70, 10); // end point of line8 (x1,y1) -  back to start
ct8.lineWidth = 3;
ct8.stroke();
</script>
