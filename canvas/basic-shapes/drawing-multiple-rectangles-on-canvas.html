Drawing Multiple Rectangles on Canvas

Create rectangles using rect() method

<canvas id=canvRect width='300' height='200'></canvas>

<script>
var canvas = document.getElementById("canvRect");
var cR = canvas.getContext("2d");

// outer rectangle
cR.lineWidth = "8";
cR.strokeStyle = "#BB8805";
cR.rect(10, 10, 280, 180);
cR.stroke();

// middle rectangle
cR.beginPath();
cR.fillStyle = 'rgba(116,219,45,0.5)';
cR.lineWidth = "5";
cR.strokeStyle = "green";
cR.rect(35, 35, 160, 130);
cR.fill();
cR.stroke();

// inner rectangle
cR.beginPath();
cR.fillStyle = 'lightblue';
cR.lineWidth = "5";
cR.strokeStyle = "blue";
cR.rect(55, 55, 220, 90);
cR.fill();
cR.stroke();
</script>
