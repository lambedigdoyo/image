HTML5 Canvas Text

Drawing Text on the Canvas

<canvas id='canvasText' width='400' height='200'></canvas>

<script>
var canvas = document.getElementById("canvasText");
var ct = canvas.getContext("2d");

ct.font = "2.5em Arial"; // font-size, font-family
ct.fillText("Gubhug Reyot", 50, 50); // ("text", x, y) 

ct.font = '45px Arial';
ct.fillStyle= 'gold';
ct.strokeStyle = 'red';
ct.lineWidth = 1; 
ct.fillText('Lambedigdoyo', 40, 90);
ct.strokeText('Lambedigdoyo', 40, 90);

ct.font = 'bold 65px Arial'; // font-weight, font-size , font-family
ct.lineWidth = 3; 
ct.textAlign = 'center'; // center text position (horizontal: ct.textAlign = 'center' - canvas.width/2)
ct.strokeText('Text', canvas.width/2, canvas.height/1.3); // canvas.height/1.3 (vertical position)
</script>
