HTML5 Canvas Circle

Drawing circle: lineWidth, strokeStyle, fillStyle

<canvas id='cvsRect' width='400' height='200'></canvas>

<script>
var canvas = document.getElementById('canvasCircle');
var cc = canvas.getContext('2d');
// (x, y, r, start, end * Math.PI)
cc.arc(60, 50, 40, 0, 2 * Math.PI); // circle-1
cc.stroke(); 
// To create a circle, 
// set start angle to 0 and end angle to 2 * Math.PI
cc.beginPath(); // start new Path
cc.arc(200, 100, 95, 0, 2 * Math.PI); // circle-2
cc.lineWidth = '5';
cc.strokeStyle = 'blue';
cc.stroke(); 

cc.beginPath(); // start new Path
cc.lineWidth = '5';
cc.strokeStyle = 'gold';
cc.fillStyle = 'red';
cc.arc(200, 100, 90, 0, 2 * Math.PI); // circle-3
cc.fill();
cc.stroke();
</script>
