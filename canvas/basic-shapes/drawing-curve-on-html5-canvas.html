HTML5 Canvas Curve

Drawing Curve on HTML5 Canvas (The arch() Method)

<canvas id='canvasCurve' width='400' height='200'></canvas>

<script>
/* The upper half circle starts at 3.14 (1*Math.PI), because the circle is 2*Math.PI (2 * 3.14) */
var canvas = document.getElementById('canvasCurve');
var cur = canvas.getContext('2d');
// Math.PI = 3.14, cirlcle = 2 * Math.PI 
// cur.arc(190, 100, 90, 0, Math.PI); same with 
// cur.arc(190, 100, 90, 0, 1 * Math.PI);

cur.lineWidth = 3; // curve-1
cur.arc(50, 50, 40, 0, Math.PI); 
cur.stroke();

cur.beginPath(); // curve-2
cur.arc(315, 50, 40, 1.57, 1.5 * Math.PI);
cur.stroke();

cur.beginPath(); // curve-3
cur.arc(310, 175, 40, 3.14, 2 * Math.PI);
cur.stroke();

cur.beginPath(); // curve-4
cur.arc(50, 150, 40, 4.71, 0.5 * Math.PI);
cur.stroke();

cur.beginPath(); // curve-5 (blue)
cur.lineWidth = 8;
cur.fillStyle = 'yellow';
cur.strokeStyle = 'blue';
cur.arc(180, 100, 90, 3.14, 2 * Math.PI);
cur.fill();
cur.stroke();

cur.beginPath(); //curve-6 (red)
cur.lineWidth = 4;
cur.fillStyle = 'pink';
cur.strokeStyle = 'red';
cur.arc(190, 100, 90, 0, Math.PI); // same as with cur.arc(190, 100, 90, 0, Math.PI);
cur.fill();
cur.stroke();
</script>
