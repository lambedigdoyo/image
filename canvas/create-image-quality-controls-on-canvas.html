Create Image Quality Controls on Canvas

Using input type number as image quality control on the canvas

Paste Image Link here: 
Paste Image Link here: 
  <input type='text' id='imageLink' value='https://raw.githubusercontent.com/lambedigdoyo/lib/main/images/merntari-pagi.jpg' style="width: calc(100% - 260px); margin: 10px 2px" />
  <input type='number' id='inputScale' min='0.05' max='1.0' step='0.05' value='0.5' style="width: 55px; margin: 10px 2px" />
  
  <div class='imageBox'><img id='imagePreview' /></div>

<script>
var canvas = document.createElement('canvas');
var ctx = canvas.getContext("2d");
var inputScale = document.getElementById('inputScale');
var setQuality = ()=> {
var image = new Image();
  image.setAttribute('crossOrigin', 'anonymous');
  image.src = document.getElementById('imageLink').value;
  image.onload = ()=> {
    // get the scale
    var quality = inputScale.value;
    canvas.width = image.naturalWidth;
    canvas.height = image.naturalHeight;
    ctx.drawImage(image, 0, 0); // (image, x, y)
    var base64Data = canvas.toDataURL('image/jpeg', + quality);
    document.getElementById("code").value = base64Data;
    document.getElementById("imagePreview").src = base64Data;
  }
}
imageLink.oninput = setQuality;
inputScale.oninput = setQuality;
setQuality();
</script>
