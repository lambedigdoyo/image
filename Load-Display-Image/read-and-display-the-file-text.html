<script>
var blob = new Blob(["Some cookies are necessary for our website and services to function properly."], { type: "text/plain" });
url = window.URL.createObjectURL(blob); 

document.getElementById('ifr').src = url;
document.getElementById('ifr').onload = ()=> {
  URL.revokeObjectURL(url); // free memory
}
  
var reader1 = new FileReader();
var reader2 = new FileReader();
reader1.onload = ()=> {
  document.getElementById('p-result').innerHTML = window.atob(reader1.result.split(',')[1]);
}
reader2.onload = ()=> {
  document.getElementById('textResult2').value = reader2.result;
}
reader1.readAsDataURL(blob);
reader2.readAsText(blob);
</script>
